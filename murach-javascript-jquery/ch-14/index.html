While loop with break statement

var number;
while (true) {
    number = parseInt(prompt("Enter a number from 1 to 10."));
    if (isNaN(number) || number < 1 || number > 10) {
        document.write("Invalid entry. Please try again.");
    } else {
        break;
    }   
}
document.write(number);

While loop with continue statement

var sum = 0;
var number = 0;
while (number <= 40) {
    number++;
    if (number % 5 !== 0) {
        continue;
    }   
    sum += number;
}
document.write(sum);

Try-catch statements

var calculateFV = function( investment, rate, years ) {
    if ( isNaN(investment) || investment <= 0 ) {
        throw new Error("calculateFV requires an investment amount greater than zero." );
    }
    if ( isNaN(rate) || rate <= 0 ) {
        throw new Error("calculateFV requires an annual rate greater than zero." );
    }
    if ( isNaN(years) || years <= 0 ) {
        throw new Error("calculateFV requires a years entry greater than zero." );
    }

    var futureValue = investment;

    for ( var i = 1; i <= years; i++ ) {
        futureValue += futureValue * rate / 100;
    }
    return futureValue.toFixed(2);
};
try {
    $("future_value").text = calculateFV(investment, rate, years);
}    
catch(error) {
    document.write(error.name + ": " + error.message);
}
finally {
    $("investment".focus();)  
}

A function that generates a random number

var getRandomNumber = function(max) {
    var random;
    if (!isNaN(max)){
        // value >= 0.0 and < 1.0
        random = Math.random();
        
        // value is an integer between 0 and max
        random = Math.floor(random * max);
        
        // value is an integer between 1 and max
        random = random + 1;
        }
        // if max is not a number, will return undefined
        return random;  
    };
    
    var random_number = getRandomNumber(100);

Trimming strings

var ltrim = function(text) {
    var start = 0;
    while (text.charAt(start) == " ") {
        start++;
    }
    return text.substring(start);
};
var result = ltrim("   JavaScript and jQuery");

var rtrim = function(text) {
    var end = text.length - 1;
    while (text.charAt(end) == " ") {
        end--;
    }
    return text.substring(0, end + 1);
};
var result = rtrim("JavaScript and jQuery    ");

var trim = function(text) {
    return ltrim(rtrim(text));
};
var result = trim("   JavaScript and jQuery    ");

Comparing strings

var equalIgnoreCase = function(text1, text2) {
    return text1.toLowerCase() == text2.toLowerCase();
};
  
if (equalIgnoreCase("JavaScript", "javascript")) {
    document.write("Strings are the same."); 
}
